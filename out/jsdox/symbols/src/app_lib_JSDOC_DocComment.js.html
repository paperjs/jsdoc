1 /**
  2 	Create a new DocComment. This takes a raw documentation comment,
  3 	and wraps it in useful accessors.
  4 	@class Represents a documentation comment object.
  5 	@constructor
  6 */ 
  7 JSDOC.DocComment = function(/**String*/comment) {
  8 	if (!comment) comment = "/** @desc undocumented */";
  9 	
 10 	this.src = JSDOC.DocComment.unwrapComment(comment);
 11 	
 12 	this.meta = "";
 13 	if (this.src.indexOf("#") == 0) {
 14 		this.meta = this.src.substring(1, 3);
 15 		this.src = this.src.substring(3);
 16 	}
 17 	
 18 	this.fixDesc();
 19 	
 20 	if (defined(JSDOC.PluginManager)) {
 21 		JSDOC.PluginManager.run("onDocCommentSrc", this);
 22 	}
 23 	
 24 	this.src = JSDOC.DocComment.shared+"\n"+this.src;
 25 	
 26 	this.tagTexts = this.src.split(/(^|[\r\n])\s*@/)
 27 		.filter(function(el){return el.match(/\S/)});
 28 	
 29 	/**
 30 		The tags found in the comment.
 31 		@type JSDOC.DocTag[]
 32 	*/
 33 	this.tags =
 34 		this.tagTexts.map(function(el){return new JSDOC.DocTag(el)});
 35 	
 36 	if (defined(JSDOC.PluginManager)) {
 37 		JSDOC.PluginManager.run("onDocCommentTags", this);
 38 	}
 39 }
 40 
 41 /**
 42 	If no @desc tag is provided, this function will add it.
 43 */
 44 JSDOC.DocComment.prototype.fixDesc = function() {
 45 	if (this.meta && this.meta != "@+") return;
 46 	if (/^\s*[^@\s]/.test(this.src)) {				
 47 		this.src = "@desc "+this.src;
 48 	}
 49 }
 50 
 51 /**
 52 	Remove slash-star comment wrapper from a raw comment string.
 53 	@type String
 54 */
 55 JSDOC.DocComment.unwrapComment = function(/**String*/comment) {
 56 	var unwrapped = comment.replace(/(^\/\*\*|\*\/$)/g, "").replace(/^\s*\* ?/gm, "");
 57 	return unwrapped;
 58 }
 59 
 60 /**
 61 	Provides a printable version of the comment.
 62 	@type String
 63 */
 64 JSDOC.DocComment.prototype.toString = function() {
 65 	return this.src;
 66 }
 67 
 68 /**
 69 	Given the title of a tag, returns all tags that have that title.
 70 	@type JSDOC.DocTag[]
 71 */
 72 JSDOC.DocComment.prototype.getTag = function(/**String*/tagTitle) {
 73 	return this.tags.filter(function(el){return el.title == tagTitle});
 74 }
 75 
 76 JSDOC.DocComment.shared = "";
