1 /**
  2 	@constructor
  3 */
  4 JSDOC.DocTag = function(src) {
  5 	this.title = "";
  6 	this.type = "";
  7 	this.name = "";
  8 	this.desc = "";
  9 	
 10 	if (defined(src)) {
 11 		var parts = src.match(/^(\S+)(?:\s+\{\s*([\S\s]+?)\s*\})?\s*([\S\s]*\S)?/);
 12 		
 13 		if (defined(parts) && parts != null) {
 14 		this.title = (parts[1].toLowerCase() || "");
 15 		this.type = (parts[2] || "");
 16 	
 17 		if (this.type) this.type = this.type.replace(/\s*(,|\|\|?)\s*/g, ", ");
 18 		this.desc = (parts[3] || "");
 19 		
 20 		// should be @type foo but we'll accept @type {foo} too
 21 		if (this.title == "type") {
 22 			if (this.type) this.desc = this.type;
 23 			
 24 			// should be @type foo, bar, baz but we'll accept @type foo|bar||baz too
 25 			if (this.desc) {
 26 				this.desc = this.desc.replace(/\s*(,|\|\|?)\s*/g, ", ");
 27 			}
 28 		}
 29 
 30 		if (this.desc) {
 31 			if (this.title == "param") { // long tags like {type} [name] desc
 32 				var m = this.desc.match(/^\s*\[([a-zA-Z0-9.$_]+)(?:\s*=([^\]]+))?\](?:\s+\{\s*([\S\s]+?)\s*\})?(?:\s+([\S\s]*\S))?/);
 33 				if (m) { // optional parameter
 34 					this.isOptional = true; // bracketed name means optional
 35 					this.name = (m[1] || "");
 36 					this.defaultValue = (m[2] || undefined);
 37 					this.type = (m[3] || this.type);
 38 					this.desc = (m[4] || "");
 39 				}
 40 				else { // required parameter
 41 					var m = this.desc.match(/^\s*([a-zA-Z0-9.$_]+)(?:\s+\{\s*([\S\s]+?)\s*\})?(?:\s+([\S\s]*\S))?/);
 42 					if (m) {
 43 						this.isOptional = false;
 44 						this.name = (m[1] || "");
 45 						this.type = (m[2] || this.type);
 46 						this.desc = (m[3] || "");
 47 					}
 48 				}
 49 			}
 50 			else if (this.title == "property" || this.title == "requires") {
 51 				var m = this.desc.match(/^\s*([a-zA-Z0-9.$_]+)(?:\s+([\S\s]*\S))?/);
 52 				if (m) {
 53 					this.name = (m[1] || "");
 54 					this.desc = (m[2] || "");
 55 				}
 56 			}
 57 			else if (this.title == "config") {
 58 				var m = this.desc.match(/^\s*\[([a-zA-Z0-9.$_]+)(?:\s*=([^\]]+))?\](?:\s+\{\s*([\S\s]+?)\s*\})?(?:\s+([\S\s]*\S))?/);
 59 				if (m) { // optional parameter
 60 					this.isOptional = true; // bracketed name means optional
 61 					this.name = (m[1] || "");
 62 					this.defaultValue = (m[2] || undefined);
 63 					this.type = (m[3] || this.type);
 64 					this.desc = (m[4] || "");
 65 				}
 66 				else { // required parameter
 67 					m = this.desc.match(/^\s*([a-zA-Z0-9.$_]+)(?:\s+\{\s*([\S\s]+?)\s*\})?(?:\s+([\S\s]*\S))?/);
 68 					if (m) {
 69 						this.isOptional = false;
 70 						this.name = (m[1] || "");
 71 						this.type = (m[2] || this.type);
 72 						this.desc = (m[3] || "");
 73 					}
 74 				}
 75 			}
 76 		}
 77 		}
 78 	}
 79 }
 80 
 81 JSDOC.DocTag.prototype.toString = function() {
 82 	return this.desc;
 83 }
