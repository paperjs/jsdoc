  7 /**
  8  * Turn a path into just the name of the file.
  9  */
 10 JSDOC.Util.fileName = function(path) {
 11 	var nameStart = Math.max(path.lastIndexOf("/")+1, path.lastIndexOf("\\")+1, 0);
 12 	return path.substring(nameStart);
 13 }
 14 
 15 /**
 16  * Turn a path into just the directory part.
 17  */
 18 JSDOC.Util.dir = function(path) {
 19 	var nameStart = Math.max(path.lastIndexOf("/")+1, path.lastIndexOf("\\")+1, 0);
 20 	return path.substring(0, nameStart-1);
 21 }

 23 /**
 24  * Get commandline option values.
 25  * @static
 26  * @param {Array} args Commandline arguments. Like ["-a=xml", "-b", "--class=new", "--debug"]
 27  * @param {object} optNames Map short names to long names. Like {a:"accept", b:"backtrace", c:"class", d:"debug"}.
 28  * @return {object} Short names and values. Like {a:"xml", b:true, c:"new", d:true}
 29  */
 30 JSDOC.Util.getOptions = function(args, optNames) {
 31 	var opt = {"_": []}; // the unnamed option allows multiple values
 32 	for (var i = 0; i < args.length; i++) {
 33 		var arg = new String(args[i]);
 34 		var name;
 35 		var value;
 36 		if (arg.charAt(0) == "-") {
 37 			if (arg.charAt(1) == "-") { // it's a longname like --foo
 38 				arg = arg.substring(2);
 39 				var m = arg.split("=");
 40 				name = m.shift();
 41 				value = m.shift();
 42 				if (typeof value == "undefined") value = true;
 43 				
 44 				for (var n in optNames) { // convert it to a shortname
 45 					if (name == optNames[n]) {
 46 						name = n;
 47 					}
 48 				}
 49 			}
 50 			else { // it's a shortname like -f
 51 				arg = arg.substring(1);
 52 				var m = arg.split("=");
 53 				name = m.shift();
 54 				value = m.shift();
 55 				if (typeof value == "undefined") value = true;
 56 				
 57 				for (var n in optNames) { // find the matching key
 58 					if (name == n || name+'[]' == n) {
 59 						name = n;
 60 						break;
 61 					}
 62 				}
 63 			}
 64 			if (name.match(/(.+)\[\]$/)) { // it's an array type like n[]
 65 				name = RegExp.$1;
 66 				if (!opt[name]) opt[name] = [];
 67 			}
 68 			
 69 			if (opt[name] && opt[name].push) {
 70 				opt[name].push(value);
 71 			}
 72 			else {
 73 				opt[name] = value;
 74 			}
 75 		}
 76 		else { // not associated with any optname
 77 			opt._.push(args[i]);
 78 		}
 79 	}
 80 	return opt;
 81 }
