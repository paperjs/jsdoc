1 /**
  2 	@constructor
  3 */
  4 JSDOC.JsDoc = function(opt) {
  5 	if (opt.constructor === Array) {
  6 		JSDOC.opt = JSDOC.Util.getOptions(opt, {d:'directory', t:'template', r:'recurse', x:'ext', p:'private', a:'allfunctions', A:'Allfunctions', e:'encoding', o:'out', h:'help', 'D[]':'define'});
  7 	}
  8 	else JSDOC.opt = opt;
  9 
 10 	if (!JSDOC.opt.e) JSDOC.opt.e = "utf-8";
 11 	JSDOC.IO.setEncoding(JSDOC.opt.e);
 12 
 13 	JSDOC.opt.srcFiles = this.getSrcFiles();
 14 	this.symbolGroup = new JSDOC.SymbolGroup(this.getSymbols());
 15 }
 16 
 17 /**
 18 	Lazy retrieval of source file list, only happens when requested, only once.
 19  */
 20 JSDOC.JsDoc.prototype.getSrcFiles = function() {
 21 	if (this.srcFiles) return this.srcFiles;
 22 	var srcFiles = [];
 23 	var ext = ["js"];
 24 	if (JSDOC.opt.x) ext = JSDOC.opt.x.split(",").map(function(x) {return x.toLowerCase()});
 25 	
 26 	function isJs($) {
 27 		var thisExt = $.split(".").pop().toLowerCase();
 28 		return (ext.indexOf(thisExt) > -1); // we're only interested in files with certain extensions
 29 	}
 30 	
 31 	for (var i = 0; i < JSDOC.opt._.length; i++) {
 32 		srcFiles = srcFiles.concat(
 33 			JSDOC.IO.ls(JSDOC.opt._[i], JSDOC.opt.r).filter(isJs)
 34 		);
 35 	}
 36 	
 37 	this.srcFiles = srcFiles;
 38 	return this.srcFiles;
 39 }
 40 
 41 JSDOC.JsDoc.prototype.getSymbols = function() {
 42 	if (this.symbols) return this.symbols;
 43 	var symbols = [];
 44 
 45 	for (var i = 0, l = this.srcFiles.length; i < l; i++) {
 46 		var srcFile = this.srcFiles[i];
 47 		
 48 		try {
 49 			var src = JSDOC.IO.readFile(srcFile);
 50 		}
 51 		catch(e) {
 52 			print("oops: "+e.message);
 53 		}
 54 
 55 		var tr = new JSDOC.TokenReader();
 56 		var tokens = tr.tokenize(new JSDOC.TextStream(src));
 57 
 58 		symbols = symbols.concat(
 59 			JSDOC.Parser.parse(
 60 				new JSDOC.TokenStream(tokens),
 61 				srcFile
 62 			)
 63 		);
 64 	}
 65 	this.symbols = symbols;
 66 	return this.symbols;
 67 }
