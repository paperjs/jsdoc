 12 JSDOC.IO.saveFile = function(outDir, fileName, content) {
 13 	var out = new Packages.java.io.PrintWriter(
 14 		new Packages.java.io.OutputStreamWriter(
 15 			new Packages.java.io.FileOutputStream(outDir+FileSeparator+fileName),
 16 			JSDOC.IO.encoding
 17 		)
 18 	);
 19 	out.write(content);
 20 	out.flush();
 21 	out.close();
 22 }

 24 JSDOC.IO.readFile = function(path) {
 25 	if (!JSDOC.IO.exists(path)) {
 26 		throw new java.io.FileNotFoundException("File doesn't exist there: "+path);
 27 	}
 28 	return readFile(path, JSDOC.IO.encoding);
 29 }

 31 JSDOC.IO.copyFile = function(inFile, outDir, fileName) {
 32 	if (fileName == null) fileName = JSDOC.Util.fileName(inFile);
 33 
 34 	var inFile = new File(inFile);
 35 	var outFile = new File(outDir+FileSeparator+fileName);
 36 	
 37 	var bis = new Packages.java.io.BufferedInputStream(new Packages.java.io.FileInputStream(inFile), 4096);
 38 	var bos = new Packages.java.io.BufferedOutputStream(new Packages.java.io.FileOutputStream(outFile), 4096);
 39 	var theChar;
 40 	while ((theChar = bis.read()) != -1) {
 41 		bos.write(theChar);
 42 	}
 43 	bos.close();
 44 	bis.close();
 45 }

 47 JSDOC.IO.mkPath  = function(/**Array*/ path) {
 48 	var make = "";
 49 	for (var i = 0, l = path.length; i < l; i++) {
 50 		make += path[i]+"/";
 51 		if (! JSDOC.IO.exists(make)) {
 52 			JSDOC.IO.makeDir(make);
 53 		}
 54 	}
 55 }

 61 JSDOC.IO.ls = function(dir, recurse, allFiles, path) {
 62 	if (path === undefined) { // initially
 63 		var allFiles = [];
 64 		var path = [dir];
 65 	}
 66 	if (path.length == 0) return allFiles;
 67 	if (recurse === undefined) recurse = 1;
 68 	
 69 	dir = new File(dir);
 70 	if (!dir.directory) return [String(dir)];
 71 	var files = dir.list();
 72 	
 73 	for (var f = 0; f < files.length; f++) {
 74 		var file = String(files[f]);
 75 		if (file.match(/^\.[^\.\/\\]/)) continue; // skip dot files
 76 
 77 		if ((new File(path.join("/")+"/"+file)).list()) { // it's a directory
 78 			path.push(file);
 79 			if (path.length-1 < recurse) JSDOC.IO.ls(path.join("/"), recurse, allFiles, path);
 80 			path.pop();
 81 		}
 82 		else {
 83 			allFiles.push((path.join("/")+"/"+file).replace("//", "/"));
 84 		}
 85 	}
 86 
 87 	return allFiles;
 88 }

 90 JSDOC.IO.exists = function(path) {
 91 	file = new File(path);
 92 
 93 	if (file.isDirectory()){
 94 		return true;
 95 	}
 96 	if (!file.exists()){
 97 		return false;
 98 	}
 99 	if (!file.canRead()){
100 		return false;
101 	}
102 	return true;
103 }

105 JSDOC.IO.open = function(path, append) {
106 	var append = true;
107 	var outFile = new Packages.java.io.File(path);
108 	var out = new Packages.java.io.PrintWriter(
109 		new Packages.java.io.OutputStreamWriter(
110 			new Packages.java.io.FileOutputStream(outFile, append),
111 			JSDOC.IO.encoding
112 		)
113 	);
