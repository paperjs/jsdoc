  2 	@constructor
  3 */
  4 JSDOC.TokenStream = function(tokens) {
  5 	this.tokens = (tokens || []);
  6 	this.cursor = -1;
  7 }

  9 /**
 10 	@constructor
 11 	@private
 12 */
 13 function VoidToken(/**String*/type) {
 14 	this.toString = function() {return "<VOID type=\""+type+"\">"};
 15 	this.is = function(){return false;}
 16 }

 18 /**
 19 	@type JSDOC.Token
 20 */
 21 JSDOC.TokenStream.prototype.look = function(/**Number*/n, /**Boolean*/considerWhitespace) {
 22 	if (typeof n == "undefined") n = 0;
 23 
 24 	if (considerWhitespace == true) {
 25 		if (this.cursor+n < 0 || this.cursor+n > this.tokens.length) return {};
 26 		return this.tokens[this.cursor+n];
 27 	}
 28 	else {
 29 		var count = 0;
 30 		var i = this.cursor;
 31 
 32 		while (true) {
 33 			if (i < 0) return new JSDOC.Token("", "VOID", "START_OF_STREAM");
 34 			else if (i > this.tokens.length) return new JSDOC.Token("", "VOID", "END_OF_STREAM");
 35 
 36 			if (i != this.cursor && (this.tokens[i] === undefined || this.tokens[i].is("SPACE") || this.tokens[i].is("NEWLINE"))) {
 37 				if (n < 0) i--; else i++;
 38 				continue;
 39 			}
 40 			
 41 			if (count == Math.abs(n)) {
 42 				return this.tokens[i];
 43 			}
 44 			count++;
 45 			(n < 0)? i-- : i++;
 46 		}
 47 
 48 		return new JSDOC.Token("", "VOID", "STREAM_ERROR"); // because null isn't an object and caller always expects an object
 49 	}
 50 }

 52 /**
 53 	@type JSDOC.Token|JSDOC.Token[]
 54 */
 55 JSDOC.TokenStream.prototype.next = function(/**Number*/howMany) {
 56 	if (typeof howMany == "undefined") howMany = 1;
 57 	if (howMany < 1) return null;
 58 	var got = [];
 59 
 60 	for (var i = 1; i <= howMany; i++) {
 61 		if (this.cursor+i >= this.tokens.length) {
 62 			return null;
 63 		}
 64 		got.push(this.tokens[this.cursor+i]);
 65 	}
 66 	this.cursor += howMany;
 67 
 68 	if (howMany == 1) {
 69 		return got[0];
 70 	}
 71 	else return got;
 72 }

 74 /**
 75 	@type JSDOC.Token[]
 76 */
 77 JSDOC.TokenStream.prototype.balance = function(/**String*/start, /**String*/stop) {
 78 	if (!stop) stop = JSDOC.Lang.matching(start);
 79 	
 80 	var depth = 0;
 81 	var got = [];
 82 	var started = false;
 83 	
 84 	while ((token = this.look())) {
 85 		if (token.is(start)) {
 86 			depth++;
 87 			started = true;
 88 		}
 89 		
 90 		if (started) {
 91 			got.push(token);
 92 		}
 93 		
 94 		if (token.is(stop)) {
 95 			depth--;
 96 			if (depth == 0) return got;
 97 		}
 98 		if (!this.next()) break;
 99 	}
100 }
