  4 JSDOC.PluginManager = {
  5 }
  6 
  7 JSDOC.PluginManager.registerPlugin = function(name, handlers) {
  8 	if (!defined(JSDOC.PluginManager.plugins))
  9 		JSDOC.PluginManager.plugins = {};
 10 	
 11 	JSDOC.PluginManager.plugins[name] = handlers;
 12 }
 13 
 14 JSDOC.PluginManager.run = function(hook, target) {
 15 	for (var name in JSDOC.PluginManager.plugins) {
 16 		if (defined(JSDOC.PluginManager.plugins[name][hook])) {
 17 			JSDOC.PluginManager.plugins[name][hook](target);
 18 		}
 19 	}
 20 }
