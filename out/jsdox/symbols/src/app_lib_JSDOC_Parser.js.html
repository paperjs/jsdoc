  8 JSDOC.Parser.parse = function(/**JSDOC.TokenStream*/ts, /**String*/srcfile) {
  9 	JSDOC.Parser.symbols = [];
 10 	
 11 	JSDOC.DocComment.shared = "";
 12 	JSDOC.Symbol.setShortcuts();
 13 	JSDOC.Symbol.srcfile = (srcfile || "");
 14 	
 15 	while(ts.next()) {
 16 		if (JSDOC.Parser.findDocComment(ts)) continue;
 17 		if (JSDOC.Parser.findFunction(ts)) continue;
 18 		if (JSDOC.Parser.findVariable(ts)) continue;
 19 	}
 20 
 21 	return JSDOC.Parser.symbols;
 22 }

 24 JSDOC.Parser.findDocComment = function(/**JSDOC.TokenStream*/ts) {
 25 /*debug*///print("~~ JSDOC.Parser.findDocComment() "+ts.look());
 26 	if (ts.look().is("JSDOC")) {
 27 		var rawComment = ts.look().data;
 28 		var docComment = new JSDOC.DocComment(rawComment);
 29 
 30 		if (docComment.meta) {
 31 			if (docComment.meta == "@+") JSDOC.DocComment.shared = docComment.src;
 32 			if (docComment.meta == "@-") JSDOC.DocComment.shared = "";
 33 			if (docComment.meta == "=+") JSDOC.Symbol.setShortcuts(docComment.src);
 34 			if (docComment.meta == "=-") JSDOC.Symbol.setShortcuts();
 35 			delete ts.tokens[ts.cursor];
 36 			return true;
 37 		}
 38 		else if (docComment.getTag("overview").length > 0) {
 39 			JSDOC.Parser.symbols.push(new JSDOC.Symbol("", [], "FILE", docComment));
 40 			delete ts.tokens[ts.cursor];
 41 			return true;
 42 		}
 43 		else if (docComment.getTag("scope").length > 0) {
 44 			var scope = docComment.getTag("scope")[0].desc;
 45 			var block = new JSDOC.TokenStream(ts.balance("LEFT_CURLY"));
 46 			if (scope) {
 47 				scope = scope.replace(/\.prototype(\.|$)/g, "#");
 48 				var virtualName = docComment.getTag("name");
 49 
 50 				if (virtualName.length && (virtualName = virtualName[0].desc)) {
 51 					JSDOC.Parser.symbols.push(new JSDOC.Symbol(virtualName, [], "VIRTUAL", docComment));
 52 				}
 53 				JSDOC.Parser.onObLiteral(block, scope);
 54 			}
 55 			return true;
 56 		}
 57 		else if (docComment.getTag("name").length > 0) {
 58 			var name = docComment.getTag("name")[0].desc;
 59 			if (name) {					
 60 				JSDOC.Parser.symbols.push(new JSDOC.Symbol(name, [], "VIRTUAL", docComment));
 61 				delete ts.tokens[ts.cursor];
 62 				return true;
 63 			}
 64 		}
 65 	}
 66 	return false;
 67 }

 69 JSDOC.Parser.findFunction = function(/**JSDOC.TokenStream*/ts, /**String*/nspace) {
 70 /*debug*///print("~~ JSDOC.Parser.findFunction() "+ts.look());
 71 	if (ts.look().is("NAME")) {
 72 		var name = ts.look().data.replace(/\.prototype(\.|$)/g, "#");
 73 					
 74 		var doc = "";
 75 		var isa = null;
 76 		var body = "";
 77 		var paramTokens = [];
 78 		var params = [];
 79 		var isInner;
 80 		
 81 		// like function foo()
 82 		if (ts.look(-1).is("FUNCTION")) {
 83 			if (nspace) {
 84 				name = nspace+"-"+name;
 85 				isInner = true;
 86 			}
 87 			isa = "FUNCTION";
 88 			
 89 			if (ts.look(-2).is("JSDOC")) {
 90 				doc = ts.look(-2).data;
 91 			}
 92 			paramTokens = ts.balance("LEFT_PAREN");
 93 			body = ts.balance("LEFT_CURLY");
 94 		}
 95 		
 96 		// like foo = function() or var foo = new function()
 97 		else if (
 98 			(ts.look(1).is("ASSIGN") && ts.look(2).is("FUNCTION"))
 99 			||
100 			(ts.look(1).is("ASSIGN") && ts.look(2).is("NEW") && ts.look(3).is("FUNCTION"))
101 		) {
102 			if (nspace) {
103 				if (ts.look(-1).is("VAR")) isInner = true;
104 				name = name.replace(/^this[.#]/, (nspace+"#").replace("##", "#"));
105 			}
106 			
107 			isa = (ts.look(2).is("NEW"))? "OBJECT" : "FUNCTION";
108 			
109 			if (ts.look(-1).is("JSDOC")) {
110 				doc = ts.look(-1).data;
111 			}
112 			else if (ts.look(-1).is("VAR") && ts.look(-2).is("JSDOC")) {
113 				doc = ts.look(-2).data;
114 			}
115 			paramTokens = ts.balance("LEFT_PAREN");
116 			body = ts.balance("LEFT_CURLY");
117 			
118 			// like foo = function(n) {return n}(42)
119 			if (ts.look(1).is("LEFT_PAREN") ) {
120 				isa = "OBJECT";
121 				ts.balance("LEFT_PAREN");
122 			}
123 			
124 			if (isa == "OBJECT") {
125 				if (doc) { // we only keep these if they're documented
126 					var docComment = new JSDOC.DocComment(doc);
127 					if (!/#$/.test(name)) { // assigning to prototype of already existing symbol
128 						JSDOC.Parser.symbols.push(new JSDOC.Symbol(name, [], isa, docComment));
129 					}
130 				}
131 
132 				JSDOC.Parser.onFnBody(
133 					new JSDOC.TokenStream(body),
134 					name
135 				);
136 				return true;
137 			}
138 		}
139 
140 		if (isa && name) {
141 			if (isa == "FUNCTION") {
142 				params = JSDOC.Parser.onParamList(paramTokens);
143 			}
144 			
145 			var docComment = new JSDOC.DocComment(doc);
146 			JSDOC.Parser.symbols.push(new JSDOC.Symbol(name, params, isa, docComment));
147 			if (isInner) JSDOC.Parser.symbols[JSDOC.Parser.symbols.length-1].isInner = true;
148 			if (body) {
149 				JSDOC.Parser.onFnBody(
150 					new JSDOC.TokenStream(body),
151 					name
152 				);
153 			}
154 			return true;
155 		}
156 	}
157 	return false;
158 }

160 JSDOC.Parser.findVariable = function(/**JSDOC.TokenStream*/ts, /**String*/nspace) {
161 /*debug*///print("~~ JSDOC.Parser.findVariable() "+ts.look());
162 	if (ts.look().is("NAME") && ts.look(1).is("ASSIGN")) {
163 		
164 		// like foo = 
165 		var name = ts.look().data;
166 		if (nspace) name = name.replace(/^this./, nspace+"#");
167 		isa = "OBJECT";
168 		var doc;
169 		if (ts.look(-1).is("JSDOC")) doc = ts.look(-1).data;
170 		else if (ts.look(-1).is("VAR") && ts.look(-2).is("JSDOC")) doc = ts.look(-2).data;
171 		
172 		name = name.replace(/\.prototype(\.|$)/, "#");
173 
174 		if (doc) { // we only keep these if they're documented
175 			var docComment = new JSDOC.DocComment(doc);
176 			var isInner = (nspace && ts.look(-1).is("VAR"));
177 			
178 			if (!/#$/.test(name)) { // assigning to prototype of already existing symbol
179 				JSDOC.Parser.symbols.push(new JSDOC.Symbol(name, [], isa, docComment));
180 				if (isInner) JSDOC.Parser.symbols[JSDOC.Parser.symbols.length-1].isInner = true;
181 			}
182 		}
183 		
184 		// like foo = {
185 		if (ts.look(2).is("LEFT_CURLY")) {
186 			JSDOC.Parser.onObLiteral(
187 				new JSDOC.TokenStream(ts.balance("LEFT_CURLY")),
188 				name
189 			);
190 		}
191 		return true;
192 	}
193 	return false;
194 }

196 JSDOC.Parser.onObLiteral = function(/**JSDOC.TokenStream*/ts, /**String*/nspace) {
197 /*debug*///print("~~ JSDOC.Parser.onObLiteral with nspace: "+nspace);
198 	while (ts.look()) {
199 		if (!ts.look().is("VOID")) {
200 			if (ts.look().is("NAME") && ts.look(1).is("COLON")) {
201 				var name = nspace+((nspace.charAt(nspace.length-1)=="#")?"":".")+ts.look().data;
202 				
203 				// like foo: function
204 				if (ts.look(2).is("FUNCTION")) {
205 					var isa = "FUNCTION";
206 					var doc = "";
207 					if (ts.look(-1).is("JSDOC")) doc = ts.look(-1).data;
208 					
209 					var params = JSDOC.Parser.onParamList(ts.balance("LEFT_PAREN"));
210 					var body = ts.balance("LEFT_CURLY");
211 					
212 					// like foo: function(n) {return n}(42)
213 					if (ts.look(1).is("LEFT_PAREN")) {
214 						isa = "OBJECT";
215 						ts.balance("LEFT_PAREN");
216 					}
217 					var docComment = new JSDOC.DocComment(doc);
218 					JSDOC.Parser.symbols.push(new JSDOC.Symbol(name, params, isa, docComment));
219 					JSDOC.Parser.onFnBody(
220 						new JSDOC.TokenStream(body),
221 						name
222 					);
223 				}
224 				// like foo: {...}
225 				else if (ts.look(2).is("LEFT_CURLY")) { // another nested object literal
226 					if (ts.look(-1).is("JSDOC")) {
227 						var isa = "OBJECT";
228 						var doc = ts.look(-1).data;
229 						var docComment = new JSDOC.DocComment(doc);
230 						
231 						JSDOC.Parser.symbols.push(new JSDOC.Symbol(name, [], isa, docComment));
232 					}
233 					
234 					JSDOC.Parser.onObLiteral(
235 						new JSDOC.TokenStream(ts.balance("LEFT_CURLY")),
236 						name
237 					);
238 				}
239 				else { // like foo: 1, or foo: "one"
240 					if (ts.look(-1).is("JSDOC")) { // we only grab these if they are documented
241 						var isa = "OBJECT";
242 						var doc = ts.look(-1).data;
243 						var docComment = new JSDOC.DocComment(doc);
244 						
245 						JSDOC.Parser.symbols.push(new JSDOC.Symbol(name, [], isa, docComment));
246 					}
247 					
248 					// skip to end of RH value ignoring values like foo: bar({blah, blah}),
249 					while (!ts.look().is("COMMA")) {
250 						if (ts.look().is("LEFT_PAREN")) ts.balance("LEFT_PAREN");
251 						else if (ts.look().is("LEFT_CURLY")) ts.balance("LEFT_CURLY");
252 						else if (!ts.next()) break;
253 					}
254 				}
255 			}
256 		}
257 		if (!ts.next()) break;
258 	}
259 }

261 JSDOC.Parser.onFnBody = function(/**JSDOC.TokenStream*/ts, /**String*/nspace) {
262 /*debug*///print(">   ~~ JSDOC.Parser.onFnBody with nspace: "+nspace+" and look = "+ts.look());
263 	while (ts.look()) {
264 		if (!ts.look().is("VOID")) {
265 			if (JSDOC.Parser.findDocComment(ts)) {
266 			}
267 			else if (JSDOC.Parser.findFunction(ts, nspace)) {
268 			}
269 			else if (JSDOC.Parser.findVariable(ts, nspace)) {
270 			}
271 		}
272 		if (!ts.next()) break;
273 	}
274 }

276 JSDOC.Parser.onParamList = function(/**Array*/paramTokens) {
277 	var params = [];
278 	for (var i = 0, l = paramTokens.length; i < l; i++) {
279 		if (paramTokens[i].is("JSDOC")) {
280 			var paramType = paramTokens[i].data.replace(/(^\/\*\* *| *\*\/$)/g, "");
281 			
282 			if (paramTokens[i+1] && paramTokens[i+1].is("NAME")) {
283 				i++;
284 				params.push({type: paramType, name: paramTokens[i].data});
285 			}
286 		}
287 		else if (paramTokens[i].is("NAME")) {
288 			params.push({name: paramTokens[i].data});
289 		}
290 	}
291 	return params;
292 }
