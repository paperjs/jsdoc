  4 JSDOC.TokenReader = function(opt) {
  5 	this.keepDocs = true;
  6 	this.keepWhite = false;
  7 	this.keepComments = false;
  8 }

 10 JSDOC.TokenReader.prototype.tokenize = function(/**TokenStream*/stream) {
 11 	var tokens = [];
 12 	/**@ignore*/ tokens.last = function() { return tokens[tokens.length-1]; }
 13 
 14 	while (!stream.look().eof) {
 15 		if (this.read_mlcomment(stream, tokens)) continue;
 16 		if (this.read_slcomment(stream, tokens)) continue;
 17 		if (this.read_dbquote(stream, tokens)) continue;
 18 		if (this.read_snquote(stream, tokens)) continue;
 19 		if (this.read_regx(stream, tokens)) continue;
 20 		if (this.read_numb(stream, tokens)) continue;
 21 		if (this.read_punc(stream, tokens)) continue;
 22 		if (this.read_space(stream, tokens)) continue;
 23 		if (this.read_newline(stream, tokens)) continue;
 24 		if (this.read_word(stream, tokens)) continue;
 25 		
 26 		tokens.push(new JSDOC.Token(stream.next(), "TOKN", "UNKNOWN_TOKEN")); // This is an error case.
 27 	}
 28 	return tokens;
 29 }

 31 JSDOC.TokenReader.prototype.read_word = function(stream, tokens) {
 32 	var found = "";
 33 	while (!stream.look().eof && JSDOC.Lang.isWordChar(stream.look())) {
 34 		found += stream.next();
 35 	}
 36 	
 37 	if (found === "") {
 38 		return false;
 39 	}
 40 	else {
 41 		var name;
 42 		if ((name = JSDOC.Lang.keyword(found))) tokens.push(new JSDOC.Token(found, "KEYW", name));
 43 		else tokens.push(new JSDOC.Token(found, "NAME", "NAME"));
 44 		return true;
 45 	}
 46 }

 48 JSDOC.TokenReader.prototype.read_punc = function(stream, tokens) {
 49 	var found = "";
 50 	var name;
 51 	while (!stream.look().eof && JSDOC.Lang.punc(found+stream.look())) {
 52 		found += stream.next();
 53 	}
 54 	
 55 	if (found === "") {
 56 		return false;
 57 	}
 58 	else {
 59 		tokens.push(new JSDOC.Token(found, "PUNC", JSDOC.Lang.punc(found)));
 60 		return true;
 61 	}
 62 }

 64 JSDOC.TokenReader.prototype.read_space = function(stream, tokens) {
 65 	var found = "";
 66 	
 67 	while (!stream.look().eof && JSDOC.Lang.isSpace(stream.look())) {
 68 		found += stream.next();
 69 	}
 70 	
 71 	if (found === "") {
 72 		return false;
 73 	}
 74 	else {
 75 		if (this.collapseWhite) found = " ";
 76 		if (this.keepWhite) tokens.push(new JSDOC.Token(found, "WHIT", "SPACE"));
 77 		return true;
 78 	}
 79 }

 81 JSDOC.TokenReader.prototype.read_newline = function(stream, tokens) {
 82 	var found = "";
 83 	
 84 	while (!stream.look().eof && JSDOC.Lang.isNewline(stream.look())) {
 85 		found += stream.next();
 86 	}
 87 	
 88 	if (found === "") {
 89 		return false;
 90 	}
 91 	else {
 92 		if (this.collapseWhite) found = "\n";
 93 		if (this.keepWhite) tokens.push(new JSDOC.Token(found, "WHIT", "NEWLINE"));
 94 		return true;
 95 	}

 98 JSDOC.TokenReader.prototype.read_mlcomment = function(stream, tokens) {
 99 	if (stream.look() == "/" && stream.look(1) == "*") {
100 		var found = stream.next(2);
101 		
102 		while (!stream.look().eof && !(stream.look(-1) == "/" && stream.look(-2) == "*")) {
103 			found += stream.next();
104 		}
105 		
106 		if (/^\/\*\*[^*]/.test(found) && this.keepDocs) tokens.push(new JSDOC.Token(found, "COMM", "JSDOC"));
107 		else if (this.keepComments) tokens.push(new JSDOC.Token(found, "COMM", "MULTI_LINE_COMM"));
108 		return true;
109 	}
110 	return false;
111 }

113 JSDOC.TokenReader.prototype.read_slcomment = function(stream, tokens) {
114 	var found;
115 	if (
116 		(stream.look() == "/" && stream.look(1) == "/" && (found=stream.next(2)))
117 		|| 
118 		(stream.look() == "<" && stream.look(1) == "!" && stream.look(2) == "-" && stream.look(3) == "-" && (found=stream.next(4)))
119 	) {
120 		
121 		while (!stream.look().eof && !JSDOC.Lang.isNewline(stream.look())) {
122 			found += stream.next();
123 		}
124 		
125 		if (this.keepComments) {
126 			tokens.push(new JSDOC.Token(found, "COMM", "SINGLE_LINE_COMM"));
127 		}
128 		return true;
129 	}
130 	return false;
131 }

133 JSDOC.TokenReader.prototype.read_dbquote = function(stream, tokens) {
134 	if (stream.look() == "\"") {
135 		// find terminator
136 		var string = stream.next();
137 		
138 		while (!stream.look().eof) {
139 			if (stream.look() == "\\") {
140 				if (JSDOC.Lang.isNewline(stream.look(1))) {
141 					do {
142 						stream.next();
143 					} while (!stream.look().eof && JSDOC.Lang.isNewline(stream.look()));
144 					string += "\\\n";
145 				}
146 				else {
147 					string += stream.next(2);
148 				}
149 			}
150 			else if (stream.look() == "\"") {
151 				string += stream.next();
152 				tokens.push(new JSDOC.Token(string, "STRN", "DOUBLE_QUOTE"));
153 				return true;
154 			}
155 			else {
156 				string += stream.next();
157 			}
158 		}
159 	}
160 	return false; // error! unterminated string
161 }

163 JSDOC.TokenReader.prototype.read_snquote = function(stream, tokens) {
164 	if (stream.look() == "'") {
165 		// find terminator
166 		var string = stream.next();
167 		
168 		while (!stream.look().eof) {
169 			if (stream.look() == "\\") { // escape sequence
170 				string += stream.next(2);
171 			}
172 			else if (stream.look() == "'") {
173 				string += stream.next();
174 				tokens.push(new JSDOC.Token(string, "STRN", "SINGLE_QUOTE"));
175 				return true;
176 			}
177 			else {
178 				string += stream.next();
179 			}
180 		}
181 	}
182 	return false; // error! unterminated string
183 }

185 JSDOC.TokenReader.prototype.read_numb = function(stream, tokens) {
186 	if (stream.look() === "0" && stream.look(1) == "x") {
187 		return JSDOC.Lang.read_hex(stream, tokens);
188 	}
189 	
190 	var found = "";
191 	
192 	while (!stream.look().eof && JSDOC.Lang.isNumber(found+stream.look())){
193 		found += stream.next();
194 	}
195 	
196 	if (found === "") {
197 		return false;
198 	}
199 	else {
200 		if (/^0[0-7]/.test(found)) tokens.push(new JSDOC.Token(found, "NUMB", "OCTAL"));
201 		else tokens.push(new JSDOC.Token(found, "NUMB", "DECIMAL"));
202 		return true;
203 	}
204 }

206 JSDOC.TokenReader.prototype.read_hex = function(stream, tokens) {
207 	var found = stream.next(2);
208 	
209 	while (!stream.look().eof) {
210 		if (JSDOC.Lang.isHexDec(found) && !JSDOC.Lang.isHexDec(found+stream.look())) { // done
211 			tokens.push(new JSDOC.Token(found, "NUMB", "HEX_DEC"));
212 			return true;
213 		}
214 		else {
215 			found += stream.next();
216 		}
217 	}
218 	return false;
219 }

221 JSDOC.TokenReader.prototype.read_regx = function(stream, tokens) {
222 	if (
223 		stream.look() == "/"
224 		&& 
225 		(
226 			!tokens.last()
227 			||
228 			(
229 				!tokens.last().is("NUMB")
230 				&& !tokens.last().is("NAME")
231 				&& !tokens.last().is("RIGHT_PAREN")
232 				&& !tokens.last().is("RIGHT_BRACKET")
233 			)
234 		)
235 	) {
236 		var regex = stream.next();
237 		
238 		while (!stream.look().eof) {
239 			if (stream.look() == "\\") { // escape sequence
240 				regex += stream.next(2);
241 			}
242 			else if (stream.look() == "/") {
243 				regex += stream.next();
244 				
245 				while (/[gmi]/.test(stream.look())) {
246 					regex += stream.next();
247 				}
248 				
249 				tokens.push(new JSDOC.Token(regex, "REGX", "REGX"));
250 				return true;
251 			}
252 			else {
253 				regex += stream.next();
254 			}
255 		}
256 		// error: unterminated regex
257 	}
258 	return false;
259 }
