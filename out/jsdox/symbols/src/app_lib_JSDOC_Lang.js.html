122 JSDOC.Lang.isNumber = function(str) {
123 	return /^(\.[0-9]|[0-9]+\.|[0-9])[0-9]*([eE][+-][0-9]+)?$/i.test(str);
124 }
125 
126 JSDOC.Lang.isHexDec = function(str) {
127 	return /^0x[0-9A-F]+$/i.test(str);
128 }
129 
130 JSDOC.Lang.isWordChar = function(str) {
131 	return /^[a-zA-Z0-9$_.]+$/.test(str);
132 }

134 JSDOC.Lang.isSpace = function(str) {

136 }

138 JSDOC.Lang.isNewline = function(str) {
