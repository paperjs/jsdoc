  2 	@constructor

  4 JSDOC.Symbol = function(name, params, isa, comment) {
  5 	
  6 	comment.src = comment.src;
  7 	
  8 	this.alias = this.name = name
  9 	
 10 	if (defined(JSDOC.PluginManager)) {
 11 		JSDOC.PluginManager.run("onSymbol", this);
 12 	}
 13 	this.name = this.alias = this.alias.replace(/\.prototype\.?/g, '#');
 14 	this.params = (params || []);
 15 	
 16 	for (var i = 0; i < this.params.length; i++) {
 17 		this.params[i] = new JSDOC.DocTag("param"+((this.params[i].type)?" {"+this.params[i].type+"}":"")+" "+this.params[i].name);
 18 	}
 19 	
 20 	this.isa = (isa || "OBJECT");
 21 	this.type = "";
 22 	
 23 	this.desc = "";
 24 	this.classDesc = "";
 25 	this.memberof = "";
 26 	this.isStatic = false;
 27 	this.since = "";
 28 	this.version = "";
 29 	this.deprecated = "";
 30 	this.augments = [];
 31 	this.inherits = [];
 32 	this.inheritsFrom = [];
 33 	this.properties = [];
 34 	this.methods = [];
 35 	//this.file = {};
 36 	this.requires = [];
 37 	this.returns = [];
 38 	this.exceptions = [];
 39 	this.events = [];
 40 	this.doc = comment;
 41 	this.see = [];
 42 	this.srcfile = JSDOC.Symbol.srcfile;
 43 
 44 	// move certain data out of the tags and into the Symbol
 45 	var overviews;
 46 	if ((overviews = this.doc.getTag("overview")) && overviews.length) {
 47 		var libraries;
 48 		if ((libraries = this.doc.getTag("name")) && libraries.length) {
 49 			this.name = libraries[0].desc;
 50 		}
 51 		else {
 52 			this.name = JSDOC.Util.fileName(JSDOC.Symbol.srcfile)
 53 		}
 54 		
 55 		this.desc = overviews[0].desc;
 56 	}
 57 	else {
 58 		var since;
 59 		if ((since = this.doc.getTag("since")) && since.length) {
 60 			this.since = since[0].desc;
 61 		}
 62 		
 63 		var version;
 64 		if ((version = this.doc.getTag("version")) && version.length) {
 65 			this.version = version[0].desc;
 66 		}
 67 		
 68 		var deprecated;
 69 		if ((deprecated = this.doc.getTag("deprecated")) && deprecated.length) {
 70 			this.deprecated = deprecated[0];
 71 		}
 72 		
 73 		var see;
 74 		if ((see = this.doc.getTag("see")) && version.length) {
 75 			this.see = see;
 76 		}
 77 		
 78 		var descs;
 79 		if ((descs = this.doc.getTag("desc")) && descs.length) {
 80 			this.desc = descs.join("\n"); // multiple descriptions are concatenated into one
 81 		}
 82 		
 83 		var ns;
 84 		if ((ns = this.doc.getTag("namespace")) && ns.length) {
 85 			this.classDesc += this.desc; // desc can't apply to the constructor as there is none.
 86 			this.classDesc += ns[0].desc;
 87 			this.isa = "CONSTRUCTOR";
 88 			this.isNamespace = true;
 89 		}
 90 		
 91 		var params;
 92 		if ((params = this.doc.getTag("param")) && params.length) { // user defined params override those defined by parser
 93 			if (this.params.length == 0) {
 94 				this.params = params;
 95 			}
 96 			else {
 97 				// merge parser-defined params with user-defined params
 98 				for (var i = 0, l = this.params.length; i < l; i++) {
 99 					var thisParam = this.params[i];
100 					var userParams = params.filter(function($){return $.name == thisParam.name});
101 					
102 					if (userParams && userParams.length) {
103 						if (!userParams[0].type && thisParam.type) userParams[0].type = thisParam.type;
104 						this.params[i] = userParams[0];
105 					}
106 				}
107 			}
108 		}
109 		
110 		var constructors;
111 		if ((constructors = this.doc.getTag("constructor")) && constructors.length) {
112 			this.isa = "CONSTRUCTOR";
113 		}
114 		
115 		var classes;
116 		if ((classes = this.doc.getTag("class")) && classes.length) {
117 			if (this.doc.getTag("static").length > 0) this.isStatic = true;
118 			this.isa = "CONSTRUCTOR"; // a class tag implies a conctuctor doclet
119 			
120 			this.classDesc += classes[0].desc; // multiple class tags are concatenated
121 		}
122 		
123 		var functions;
124 		if ((functions = this.doc.getTag("function")) && functions.length) {
125 			this.isa = "FUNCTION";
126 		}
127 		
128 		var events;
129 		if ((events = this.doc.getTag("event")) && events.length) {
130 			this.isa = "FUNCTION";
131 			this.isEvent = true;
132 		}
133 		
134 		var methods;
135 		if ((functions = this.doc.getTag("method")) && functions.length) {
136 			this.isa = "FUNCTION";
137 		}
138 		
139 		var names;
140 		if ((names = this.doc.getTag("name")) && names.length) {
141 			this.name = names[0].desc;
142 		}
143 		
144 		var properties;
145 		if ((properties = this.doc.getTag("property")) && properties.length) {
146 			for (var i = 0; i < properties.length; i++) {
147 				properties[i].alias = this.alias+"."+properties[i].name;
148 				this.properties.push(properties[i]);
149 			}
150 		}
151 		
152 		var returns;
153 		if ((returns = this.doc.getTag("return")) && returns.length) {
154 			for (var i = 0; i < returns.length; i++) {
155 				this.returns.push(returns[i]);
156 			}
157 		}
158 		
159 		var exceptions;
160 		if ((exceptions = this.doc.getTag("throws")) && exceptions.length) {
161 			for (var i = 0; i < exceptions.length; i++) {
162 				this.exceptions.push(exceptions[i]);
163 			}
164 		}
165 		
166 		var requires;
167 		if ((requires = this.doc.getTag("requires")) && requires.length) {
168 			for (var i = 0; i < requires.length; i++) {
169 				this.requires.push(requires[i]);
170 			}
171 		}
172 		
173 		if (this.is("VIRTUAL")) this.isa = "OBJECT";
174 		
175 		var types;
176 		if ((types = this.doc.getTag("type")) && types.length) {
177 			this.type = (types[0].desc || ""); // multiple type tags are ignored
178 		}
179 		
180 		if (this.doc.getTag("static").length > 0) {
181 			this.isStatic = true;
182 		}
183 		
184 		if (this.doc.getTag("private").length > 0) {
185 			this.isPrivate = true;
186 		}
187 		
188 		var inherits;
189 		if ((inherits = this.doc.getTag("inherits")) && inherits.length) {
190 			for (var i = 0; i < inherits.length; i++) {
191 				this.inherits.push(inherits[i].desc);
192 			}
193 		}
194 		
195 		var augments;
196 		if ((augments = this.doc.getTag("augments")) && augments.length) {
197 			for (var i = 0; i < augments.length; i++) {
198 				this.augments.push(augments[i].desc);
199 			}
200 		}
201 		
202 		var defaults;
203 		if ((defaults = this.doc.getTag("default")) && defaults.length) {
204 			if (this.is("OBJECT")) {
205 				this.defaultValue = defaults[0];
206 			}
207 		}
208 		
209 		var memberofs;
210 		if ((memberofs = this.doc.getTag("memberof")) && memberofs.length) {
211 			this.memberof = memberofs[0].desc;
212 		}
213 		
214 		if (defined(JSDOC.PluginManager)) {
215 			JSDOC.PluginManager.run("onSymbol", this);
216 		}
217 	}
218 }

220 JSDOC.Symbol.prototype.setDescription = function(desc) {
221 	this.desc = desc;
222 }
223 JSDOC.Symbol.prototype.getDescription = function() {
224 	return this.desc;
225 }
226 
227 JSDOC.Symbol.prototype.is = function(what) {
228 	return this.isa === what;
229 }
230 
231 JSDOC.Symbol.prototype.hasTag = function(tagTitle) {
232 	for (var i = 0, l = this.doc.tags.length; i < l; i++) {
233 		if (this.doc.tags[i].title == tagTitle)
234 			return true;
235 	}
236 	return false;
237 }

239 /** Generate a comma separated list of the parameters. */
240 JSDOC.Symbol.prototype.signature = function() {
241 	var result = [];
242 	for (var i = 0; i < this.params.length; i++) {
243 		if (this.params[i].name.indexOf(".") == -1) // config information does not appear in the signature
244 			result.push(this.params[i].name);
245 	}
246 	return result.join(", ");
247 }

249 JSDOC.Symbol.prototype.inherit = function(symbol) {
250 	if (!this.hasMember(symbol.name)) {
251 		if (symbol.isa == "FUNCTION" && !symbol.isInner)
252 			this.methods.push(symbol);
253 		else if (symbol.isa == "OBJECT")
254 			this.properties.push(symbol);
255 	}
256 }
257 
258 JSDOC.Symbol.prototype.getMethods = function() {
259 	return this.methods.filter(
260 		function(method) {
261 			if (!method.isEvent) return method;
262 		}
263 	);
264 }
265 
266 JSDOC.Symbol.prototype.getProperties = function() {
267 	return this.properties;
268 }
269 
270 JSDOC.Symbol.prototype.getEvents = function() {
271 	return this.methods.filter(
272 		function(method) {
273 			if (method.isEvent) return method;
274 		}
275 	);
276 }
277 
278 JSDOC.Symbol.prototype.hasMember = function(name) {
279 	return (this.hasMethod(name) || this.hasProperty(name));
280 }
281 
282 JSDOC.Symbol.prototype.hasMethod = function(name) {
283 	for (var i = 0; i < this.methods.length; i++) {
284 		if (this.methods[i].name == name) return true
285 	}
286 	return false;
287 }
288 
289 JSDOC.Symbol.prototype.hasProperty = function(name) {
290 	for (var i = 0; i < this.properties.length; i++) {
291 		if (this.properties[i].name == name) return true
292 	}
293 	return false;
294 }
