  5 JSDOC.SymbolGroup = function(symbols) {
  6 	this.symbols = this.filterByOption(symbols, JSDOC.opt);
  7 	global = new JSDOC.Symbol("Global", [], "CONSTRUCTOR", new JSDOC.DocComment("/** BUILTIN */"));
  8 	this.symbols.push(global);
  9 	
 10 	this.fileIndex = new Hash();
 11 	this.classIndex = {};
 12 	this.nameIndex = new Hash();
 13 	this.typeIndex = new Hash();
 14 	
 15 	this.indexAll();
 16 	this.resolveMemberOf();
 17 	this.indexAll();
 18 	this.resolveNames();
 19 	this.indexAll(this.symbols);
 20 	this.resolveMembers();
 21 	this.resolveInherits();
 22 }

 24 JSDOC.SymbolGroup.prototype.getOverview = function(path) {
 25 	var overviews = this.getSymbolsByType().filter(
 26 		function(symbol) {
 27 			return (symbol.srcfile == path);
 28 		}
 29 	);
 30 	if (overviews.constructor === Array) return overviews[0];

 33 /** Apply any effects of -a -A -p (etcetera) commandline options */
 34 JSDOC.SymbolGroup.prototype.filterByOption = function(symbols, options) {			 
 35 	symbols = symbols.filter(
 36 		function(symbol) {
 37 			if (symbol.isInner) symbol.isPrivate = true;
 38 			
 39 			var keep = true;
 40 			
 41 			if (symbol.isa == "FILE") keep = true;
 42 			else if (symbol.desc == "undocumented" && !(options.a ||options.A)) keep = false;
 43 			else if (/(^|[.#-])_/.test(symbol.alias) && !options.A) keep = false;
 44 			else if (symbol.isPrivate && !options.p) keep = false;
 45 			else if (symbol.doc.getTag("ignore").length > 0) keep = false;
 46 			
 47 			if (/#$/.test(symbol.alias)) keep = false; // we don't document prototype
 48 			
 49 			if (keep) return symbol
 50 		}
 51 	);
 52 	return symbols;
 53 }

 75 JSDOC.SymbolGroup.prototype.indexAll = function() {
 76 	this.nameIndex.reset();
 77 	this.typeIndex.reset();
 78 	this.fileIndex.reset();
 79 	
 80 	// remove any symbols set to null
 81 	this.symbols = this.symbols.filter(function($){return $ != null});
 82 	
 83 	for (var i = 0, l = this.symbols.length; i < l; i++) {
 84 		this.indexSymbol(this.symbols[i]);
 85 	}
 86 }

 88 JSDOC.SymbolGroup.prototype.indexSymbol = function(symbol) {
 89 	// filename=>symbol[] map
 90 	var srcfile = symbol.srcfile;
 91 	if (srcfile) {
 92 		if (!this.fileIndex.has(srcfile)) this.fileIndex.put(srcfile, []);
 93 		this.fileIndex.get(srcfile).push(symbol);
 94 	}
 95 	
 96 	// alias=>symbol map, presumes symbol.alias are unique
 97 	if (symbol.alias) {
 98 		this.nameIndex.put(symbol.alias, symbol);
 99 	}
100 
101 	// isa=>symbol[] map
102 	if (symbol.isa) {
103 		var type = symbol.isa;
104 		if (!this.typeIndex.has(type)) this.typeIndex.put(type, []);
105 		this.typeIndex.get(type).push(symbol);
106 	}
107 }

109 JSDOC.SymbolGroup.prototype.addBuiltIn = function(isa) {
110 	if (this.getSymbol(isa)) return;
111 	var docComment = new JSDOC.DocComment("/** BUILTIN */");
112 	var builtIn = new JSDOC.Symbol(isa, [], "CONSTRUCTOR", docComment, "");
113 	builtIn.isStatic = true;
114 	this.symbols.push(builtIn);
115 	this.indexSymbol(builtIn);
116 	return builtIn;
117 }

119 JSDOC.SymbolGroup.prototype.resolveMemberOf = function(symbol) {
120 	for (var i = 0, l = this.symbols.length; i < l; i++) {
121 		var symbol = this.symbols[i];
122 		if (symbol.alias == "Global" || symbol.is("FILE")) continue;
123 		
124 		if (symbol.memberof) {
125 			symbol.memberof = symbol.memberof.replace(/\.prototype(\.|$)/g, "#");
126 			
127 			if (symbol.memberof.charAt(symbol.memberof.length-1) == "#") {
128 				symbol.alias = symbol.memberof+symbol.alias;
129 			}
130 			else symbol.alias = symbol.memberof+"."+symbol.alias;
131 		}
132 	}
133 }

135 JSDOC.SymbolGroup.prototype.resolveNames = function() {
136 	eachSymbol:
137 	for (var i = 0, l = this.symbols.length; i < l; i++) {
138 		var symbol = this.symbols[i];
139 		if (symbol.alias == "Global" || symbol.is("FILE")) continue;
140 
141 		var nameChain = new Chain(symbol.alias.split(/([#.-])/));
142 
143 		// find the constructor closest in the chain to "this"
144 		symbol.parentConstructor = "";
145 		for (var node = nameChain.last(); node !== null; node = nameChain.prev()) {
146 			var parentName = nameChain.joinLeft();
147 			if (parentName) {
148 				var parent = this.getSymbol(parentName);
149 				if (
150 					(parent && parent.isa == "CONSTRUCTOR")
151 					||
152 					(symbol.addOn = JSDOC.Lang.isBuiltin(parentName))
153 				) {
154 					symbol.parentConstructor = parentName;
155 					if (symbol.addOn) {
156 						this.addBuiltIn(parentName);
157 					}
158 					break;
159 				}
160 			}
161 		}
162 
163 		if (symbol.parentConstructor) {
164 			// constructor#blah#foo => constructor#foo
165 			var oldAlias = symbol.alias;
166 			symbol.alias = 
167 				symbol.alias.replace(
168 					new RegExp("^"+RegExp.escapeMeta(symbol.parentConstructor)+'(\\.|#)[^+]+#'), 
169 					symbol.parentConstructor+"#"
170 				);
171 			this.nameIndex.rename(oldAlias, symbol.alias);
172 
173 			var parts = symbol.alias.match(/^(.*[.#-])([^.#-]+)$/);
174 			if (parts) {
175 				symbol.memberof = parts[1];
176 				symbol.name = parts[2];
177 
178 				if (symbol.memberof) {
179 					switch (symbol.memberof.charAt(symbol.memberof.length-1)) {
180 						case '#' :
181 							symbol.isStatic = false;
182 							symbol.isInner = false;
183 						break;
184 						case '.' :
185 							symbol.isStatic = true;
186 							symbol.isInner = false;
187 						break;
188 						case '-' :
189 							symbol.isStatic = true;
190 							symbol.isInner = true;
191 						break;
192 					}
193 					symbol.memberof = symbol.memberof.substr(0, symbol.memberof.length-1);
194 				}
195 				else {
196 					symbol.isStatic = true;
197 					symbol.isInner = false;
198 				}
199 			}
200 			if (!this.classIndex[symbol.memberof]) this.classIndex[symbol.memberof] = [];
201 			this.classIndex[symbol.memberof].push(symbol);
202 		}
203 		else { // no parent constructor
204 			symbol.alias = symbol.alias.replace(/^(Global#)?([^#]+)(\.[^#.]+)#(.+)$/, "$1$2.$4");
205 			if (RegExp.$2 && RegExp.$4) symbol.name = RegExp.$2+"."+RegExp.$4;
206 
207 			if (!symbol.is("CONSTRUCTOR")) {
208 				if (symbol.alias.indexOf("#") > -1) {
209 					print("WARNING: Documentation found for instance member: "+this.symbols[i].name+", but no documentation exists for parent class.");
210 					this.symbols[i] = null;
211 					continue eachSymbol;
212 				}
213 				
214 				symbol.memberof = "Global";
215 				symbol.alias = "Global#"+symbol.name;
216 			}
217 			
218 			symbol.isStatic = true;
219 			symbol.isInner = false;
220 			global.inherit(symbol);
221 		}
222 	}
223 }

225 JSDOC.SymbolGroup.prototype.resolveMembers = function() {
226 	for (var i = 0, l = this.symbols.length; i < l; i++) {
227 		var symbol = this.symbols[i];
228 		if (symbol.is("FILE")) continue;
229 		
230 		var members = this.classIndex[symbol.alias];
231 		if (members) {
232 			for(var ii = 0, il = members.length; ii < il; ii++) {
233 				var member = members[ii];
234 				if (member.isa == "FUNCTION") {
235 					symbol.methods.push(member);
236 				}
237 				if (member.isa == "OBJECT") {
238 					symbol.properties.push(member);
239 				}
240 			}
241 		}
242 	}
243 }

245 JSDOC.SymbolGroup.prototype.resolveInherits = function() {
246 	for (var i = 0, l = this.symbols.length; i < l; i++) {
247 		var symbol = this.symbols[i];
248 		if (symbol.alias == "Global" || symbol.is("FILE")) continue;
249 		
250 		// add in inherited members
251 		for(var ii = 0, il = symbol.inherits.length; ii < il; ii++) {
252 			symbol.inherits[ii] = symbol.inherits[ii].replace(/\.prototype(\.|$)/g, "#");
253 			var inheritedMember = this.getSymbol(symbol.inherits[ii]);
254 			if (inheritedMember) {
255 				symbol.inherit(inheritedMember);
256 				symbol.inheritsFrom.push(inheritedMember.memberof);
257 			}
258 		}
259 		
260 		for(var ii = 0, il = symbol.augments.length; ii < il; ii++) {
261 			var contributer = this.getSymbol(symbol.augments[ii]);
262 			if (contributer) {
263 				symbol.inheritsFrom.push(contributer.alias);
264 				if (!isUnique(symbol.inheritsFrom)) {
265 					LOG.warn("Can't resolve inherits: Circular reference: "+symbol.alias+" inherits from "+contributer.alias+" more than once.");
266 				}
267 				else {
268 					for (var ci = 0, cl = contributer.methods.length; ci < cl; ci++)
269 						symbol.inherit(contributer.methods[ci]);
270 					for (var ci = 0, cl = contributer.properties.length; ci < cl; ci++)
271 						symbol.inherit(contributer.properties[ci]);
272 				}
273 			}
274 		}
275 	}
